app.js


angular.module('app', [
	'ngI18n',
	'ngI18nLangSelect'
])

.config(function(I18NProvider) {
		var i18nConfig = {
			langsDir: 'assets/lang/',
			langs: [{
				locale: 'en-us',
				flag: 'us',
				name: 'English'
			}, {
				locale: 'pt-pt',
				flag: 'pt',
				name: 'Portuguese'
			}]
		};
		I18NProvider.config(i18nConfig);
	})
